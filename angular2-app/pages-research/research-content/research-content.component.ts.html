import { Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
 
import { ResearchService } from '../../_services/research.service';
 
@Component({
  selector: 'app-research-content',
  templateUrl: 'research-content.component.html',
  styleUrls: ['research-content.component.scss']
})
export class ResearchContentComponent implements OnInit {
 
  /** Ids from Route Parameter **/
  private tileId: string;
  private contentId: string;
 
  /** Iframe Details **/
  private contentDetails: Object = { };
 
  /** Error Reporting **/
  private isDataLoaded: boolean = false;
  private isError: boolean = false;
  private errorDetails: Object = { };
 
  constructor( private researchService: ResearchService, private route: ActivatedRoute ) { }
 
  ngOnInit() {
 
    // Get tileId from route
    this.route.params.flatMap(params => {
 
      // Get Section Id
      let tileId = params['tileId'];
      let contentId = params['contentId'];
 
      // Add tileId and contentId to scope
      this.tileId = tileId;
      this.contentId = contentId;
 
      // Set tileId and contentId in contentDetails
      // If invalid ids are provided and page breaks, breadcrumbs will not be empty
      // This will be overwritten when actual contents are retrieved
      this.contentDetails[ 'tileName'] = tileId;
      this.contentDetails[ 'title' ] = contentId;
 
      //  Get Student Data
      return this.researchService.getContentDetails( tileId, contentId );
 
    }).subscribe(data => {
 
      // Set Content Details
      this.contentDetails = data;
 
      // Set Flags
      this.isDataLoaded = true;
      this.isError = false;
 
    }, err => {
 
      // Set Error Details
      this.errorDetails = err;
 
      // Set Flags
      this.isDataLoaded = false;
      this.isError = true;
 
    });
 
  }
 
}
