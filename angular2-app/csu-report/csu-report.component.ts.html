import { Component, OnInit } from '@angular/core';
import { StudentService } from '../_services/student.service';
import { Router, ActivatedRoute } from '@angular/router';
 
 
@Component( {
    selector: 'app-csu-report',
    templateUrl: './csu-report.component.html',
    styleUrls: [ './csu-report.component.scss' ]
} )
export class CSUReportComponent implements OnInit {
 
    private data;
    private sectionId;
    private students = [];
    private chartOptions;
 
    /** Data for Header Component **/
    private chartTitle: string = "Attrition Risk Analysis";
    private courseName: string = "";
    private weekInfo: string = "";
 
    /** Filter Component Options **/
    private filterMap = {
        low: true, medium: true, high: true
    };
 
    /** Sorting Component Options **/
    private sortByModel = 'score';
    private sortByOptions = [
        { title: 'Risk Score', model: 'score' },
        { title: 'Student Name', model: 'name' }
    ];
 
    /** Flags and Defaults **/
    private dataLoaded = false;
    private isError = false;
    private error = { };
 
    constructor( private studentService: StudentService, private route: ActivatedRoute ) {
 
    }
 
    ngOnInit() {
 
        // Get sectionId from route
        this.route.params.flatMap( params => {
            console.log( params );
 
            // Get Section Id
            let sectionId = params[ 'sectionId' ];
            this.sectionId = sectionId;
 
            //  Get Student Data
            return this.studentService.getStudentsData( sectionId );
 
        } ).subscribe( data => {
 
            // Setup data for chart and table
          console.log("DATA", data);
 
            // Count students by risk category
            let counts = { high: 0, medium: 0, low: 0 };
            data.students.forEach( student => {
                counts[ student.category ]++;
            } );
 
            // Add Metrics to Chart Options
            // Chart Options
            let chartOptions = {
                credits: false,
                title: { text: null },
                chart: { type: 'column', height: 250 },
                xAxis: {
                    tickPosition: 'inside',
                    tickLength: 5,
                    tickColor: "#3f3f3f",
                    lineColor: "#3f3f3f",
                    type: 'category',
                    labels: {
                        style: { fontWeight: 'bold' },
                    },
                    categories: [
                        'High Risk',
                        'Medium Risk',
                        'Low Risk'
                    ]
                },
                yAxis: {
                    gridLineWidth: 0,
                    minorGridLineWidth: 0,
                    tickPosition: 'inside',
                    maxPadding: 0,
                    endOnTick: true,
                    tickWidth: 1,
                    tickColor: "#3f3f3f",
                    lineColor: "#3f3f3f",
                    lineWidth: 1,
                    title: { text: 'Number of Students' }
                },
                tooltip: {
                    formatter: function () {
                        return 'Students: <b>' + this.y + '</b>';
                    }
                },
                series: [ {
                    showInLegend: false,
                    data: [
                        { y: counts.high, color: '#F45656' },
                        { y: counts.medium, color: '#EEE140' },
                        { y: counts.low, color: '#AFC917' }
                    ]
                } ]
            };
 
            // Extract Course and Week Information
            this.courseName = data.metadata.course + ", " + data.metadata.section;
            this.weekInfo = "Predictions last updated on " + data.metadata.date;
 
            // Add Student Data and Options to the Scope
            this.data = data;
            this.students = data.students.slice();
            this.chartOptions = chartOptions;
 
            // Default SortBy
            this.sortBy( this.sortByModel );
 
            // Data Loaded
            this.dataLoaded = true;
 
        }, err => {
              this.isError = true;
              this.error = err;
          }
        );
    }
 
    sortBy( model ) {
 
        // Keep track of updated sortByModel
        this.sortByModel = model;
 
        // Sort
        this.students.sort( ( a, b ) => {
            if ( model === 'score' ) {
                return a[ model ] < b[ model ] ? 1 : -1;
            } else {
                return a[ model ].toLowerCase() > b[ model ].toLowerCase() ? 1 : -1;
            }
        } );
 
    }
 
    toggleFilterCategory( target ) {
 
        // Keep Track of Checked Values
        this.filterMap[ target.value ] = target.checked;
 
        // Filter Students from Original Dataset
        this.students = this.data.students.filter( student => this.filterMap[ student.category ] );
 
        // Sort by latest sortModel selected
        this.sortBy( this.sortByModel );
 
    }
 
 
}
