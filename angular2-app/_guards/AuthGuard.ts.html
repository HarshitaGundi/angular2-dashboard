import {Injectable} from '@angular/core';
import {Router, RouterStateSnapshot, ActivatedRouteSnapshot, CanActivate} from '@angular/router';
import {UserService} from '../_services/user.service';
 
@Injectable()
export class AuthGuard implements CanActivate {
 
    constructor(private router: Router, private userService: UserService) {
    }
 
    /**
     * Uses UserService#isLoggedIn() to determine whether to activate a path or not
     *
     * @returns {any|boolean}
     */
    canActivate(route: ActivatedRouteSnapshot, state: RouterStateSnapshot) {
 
        // Use UserService to check if user is loggedIn
        let isLoggedIn = this.userService.isLoggedIn();
 
        // If token is not found or expired navigate to login page
        // Use Angular2 navigation
        if( !isLoggedIn ){
          this.router.navigate( [""], { queryParams: { redirectUrl: state.url } } );
        }
 
        // When token is not null and is not expired return true
        return isLoggedIn;
 
    }
}
