import { RouterModule, Routes } from '@angular/router';
import { CSUReportComponent } from './pages-prototypes/csu-report/csu-report.component';
import { AuthGuard } from './_guards/AuthGuard';
import { PrototypesComponent } from './pages-prototypes/prototypes/prototypes.component';
import { LoginComponent } from './login/login.component';
import { CSUSectionComponent }  from './pages-prototypes/csu-section/csu-section.component';
import { ResearchComponent }  from './pages-research/research/research.component'
import { ResearchListComponent } from './pages-research/research-list/research-list.component';
import { ResearchContentComponent } from './pages-research/research-content/research-content.component';
 
const appRoutes: Routes = [
 
    // Default path is a Login Page
    { path: '', component: LoginComponent },
 
    // Login Page
    { path: 'login', component: LoginComponent },
 
    // Prototype Area
    { path: 'proto', component: PrototypesComponent, canActivate: [ AuthGuard ] },
    { path: 'proto/csu/section', component: CSUSectionComponent, canActivate: [ AuthGuard ] },
    { path: 'proto/csu/section/:sectionId', component: CSUReportComponent, canActivate: [ AuthGuard ] },
 
    // Research Area
    { path: 'research', component: ResearchComponent, canActivate: [ AuthGuard ] },
    { path: 'research/:tileId', component: ResearchListComponent, canActivate: [ AuthGuard ] },
    { path: 'research/:tileId/:contentId', component: ResearchContentComponent, canActivate: [ AuthGuard ] },
 
    // Since we have updated the routerLink from /home to /proto
    // Make /home redirect to /proto just in case anybody has it bookmarked
    { path: 'home', redirectTo: 'proto' },
 
    // All rest of the paths should redirect to login
    { path: '**', redirectTo: 'login' }
 
];
 
export const routes = RouterModule.forRoot( appRoutes );
